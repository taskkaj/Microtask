  <script>
    // Demo JS (client-side only)
    const taskListEl = document.getElementById('taskList');
    const adminListEl = document.getElementById('adminList');
    const taskCountEl = document.getElementById('taskCount');
    const walletEl = document.getElementById('wallet');
    const yearEl = document.getElementById('year');
    yearEl.textContent = new Date().getFullYear();

    let wallet = 0;
    let tasks = [];

    function renderTasks(filter='all'){
      taskListEl.innerHTML='';
      const filtered = tasks.filter(t => filter==='all' || t.category===filter);
      taskCountEl.textContent = filtered.length;
      filtered.forEach((t,i)=>{
        const div = document.createElement('div'); div.className='task';
        div.innerHTML = `<div style="display:flex;justify-content:space-between;align-items:center"><strong>${escapeHtml(t.title)}</strong><div class="muted">৳${t.price} x ${t.qty}</div></div>
          <p class="muted">${escapeHtml(t.desc)}</p>
          <div style="display:flex;gap:8px;justify-content:flex-end"><button class="btn" onclick="acceptTask(${i})">গ্রহণ করুন</button></div>`;
        taskListEl.appendChild(div);
      });
      renderAdmin();
    }

    function renderAdmin(){
      adminListEl.innerHTML='';
      tasks.forEach((t,i)=>{
        const d = document.createElement('div'); d.style.padding='8px'; d.style.borderBottom='1px solid #eef2f4';
        d.innerHTML = `<div style="display:flex;justify-content:space-between;align-items:center"><div><strong>${escapeHtml(t.title)}</strong> <div class='muted'>${escapeHtml(t.category)} • ৳${t.price} • qty ${t.qty}</div></div>
          <div style="display:flex;gap:8px"><button onclick="togglePublish(${i})">${t.published? 'Unpublish': 'Publish'}</button><button onclick="removeTask(${i})">Remove</button></div></div>`;
        adminListEl.appendChild(d);
      });
    }

    function postTask(e){
      e.preventDefault();
      const title = document.getElementById('title').value.trim();
      const desc = document.getElementById('desc').value.trim();
      const category = document.getElementById('category').value;
      const price = Number(document.getElementById('price').value);
      const qty = Number(document.getElementById('qty').value);
      if(!title||!desc||price<=0||qty<=0) return alert('সঠিক তথ্য দিন');

      // Prohibited quick-check for suspicious social tasks
      if(category==='social' && /subscrib|like|follow|fake|bot/i.test(title+desc)){
        if(!confirm('সোশ্যাল ক্যাটাগরির কাজ হতে পারে প্ল্যাটফর্ম নীতির বিরুদ্ধে। আপনি নিশ্চিত?')) return;
      }

      tasks.push({title,desc,category,price,qty,published:true});
      document.getElementById('postForm').reset();
      renderTasks(document.getElementById('categoryFilter').value);
      alert('কাজ পোস্ট করা হয়েছে (ডেমো — UI only)');
    }

    function acceptTask(index){
      const t = tasks[index];
      if(!t || !t.published) return alert('টাস্ক পাওয়া যায়নি');
      if(wallet < t.price) return alert('ব্যালেন্স পর্যাপ্ত নয় — আগে ব্যালেন্স বাড়ান');
      if(!confirm(`আপনি '${t.title}' গ্রহণ করতে চান?`)) return;
      // Deduct price and simulate completion
      wallet -= t.price; updateWallet();
      t.qty -= 1; if(t.qty<=0) tasks.splice(index,1);
      renderTasks(document.getElementById('categoryFilter').value);
      alert('টাস্ক গ্রহণ করা হয়েছে — ডেমো মোডে তাত্ক্ষণিক সমাপ্তির ওয়্যার-ফ্লো দেখানো হয়েছে');
    }

    function addFundsPrompt(){
      const amt = Number(prompt('কত টাকা যোগ করবেন? (ডেমো) — সংখ্যা লিখুন'));
      if(amt && amt>0){ wallet += amt; updateWallet(); }
    }
    function updateWallet(){ walletEl.textContent = wallet; }

    function applyFilter(){ const v = document.getElementById('categoryFilter').value; renderTasks(v); }

    function openAdmin(){ document.getElementById('adminModal').style.display='flex'; }
    function closeAdmin(){ document.getElementById('adminModal').style.display='none'; }

    function togglePublish(i){ tasks[i].published = !tasks[i].published; renderTasks(document.getElementById('categoryFilter').value); }
    function removeTask(i){ if(confirm('মুছে ফেলতে চান?')){ tasks.splice(i,1); renderTasks(document.getElementById('categoryFilter').value); } }

    // small helper
    function escapeHtml(s){ return s.replaceAll('&','&amp;').replaceAll('<','&lt;').replaceAll('>','&gt;'); }

    // seed demo tasks
    tasks.push({title:'অ্যাপ বাগ রিপোর্ট করুন',desc:'প্রতিটি বাগ রিপোর্টের জন্য নিদের্শনা মেনে স্ক্রিনশট দিন।',category:'testing',price:20,qty:5,published:true});
    tasks.push({title:'প্রোডাক্ট ছবিতে ট্যাগিং',desc:'ছবির বস্তুগুলোকে লেবেল করুন (data labeling)।',category:'data',price:15,qty:10,published:true});
    tasks.push({title:'শিরোনাম লিখুন — ছোট ক্লিপ',desc:'ভিডিওর জন্য 5টি সম্ভাব্য শিরোনাম লিখুন।',category:'creative',price:30,qty:3,published:true});

    renderTasks(); updateWallet();
  </script>
